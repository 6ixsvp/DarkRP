-- Staff Job Category
DarkRP.createCategory {
    name = "Staff",
    categorises = "jobs",
    startExpanded = true,
    color = Color(50, 50, 50), -- A darker color for contrast
    sortOrder = 999,
    canSee = function(ply) 
         return true 
    end,
}
